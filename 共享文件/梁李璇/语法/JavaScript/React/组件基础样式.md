# 行内样式控制

```
import React, { useState } from 'react';//两个都要导入
function App() {
  const style = {
    color:'red',
    fontSize:'50px'
  }
  return (
    <div className="App">
        {/* 行内样式控制 */}
      <span style={{ color: 'red',fontSize:'50px' }}>this is span</span>
      <span style={style}>this is span</span>
     </div>
   );
}
```

# 通过class类名控制

```
import React, { useState } from 'react';//两个都要导入
function App() {
  import './index.css';//从index文件中导入样式描述
  return (
    <div className="App">
        {/*通过class类名控制*/}
      <span className="foo">this is class foo</span>
     </div>
   );
}
```

```
/* index.css文件中 */
.foo{
    color: blue;
}
```
