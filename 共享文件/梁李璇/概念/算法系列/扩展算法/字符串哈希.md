#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

const int MAXN = 10001;
const int BASE = 499;

long long nums[MAXN];
int n;

int v(char c) {
    if (c >= '0' && c <= '9') return c - '0' + 1;
    else if (c >= 'A' && c <= 'Z') return c - 'A' + 11;
    else return c - 'a' + 37;
}

long long value(const char* s) {
    long long ans = v(s[0]);
    for (int i = 1; s[i]; i++) {
        ans = ans * BASE + v(s[i]);
    }
    return ans;
}

int cnt() {
    sort(nums, nums + n);
    int ans = 1;
    for (int i = 1; i < n; i++) {
        if (nums[i] != nums[i - 1]) {
            ans++;
        }
    }
    return ans;
}

int main() {
    scanf("%d\n", &n);
    char s[200];
    for (int i = 0; i < n; i++) {
        fgets(s, sizeof(s), stdin);
        s[strcspn(s, "\n")] = '\0';  // 去掉换行符
        nums[i] = value(s);
    }
    printf("%d\n", cnt());
    return 0;
}
