## 1.快速排序qsort()（从c语言保留）

```c++
#include<cstdlib>
//头文件

#include<iostream>
using namespace std;
//升序
int compare1(const void* a, const void* b)
{
    return *(int*)a - *(int*)b;
}
//降序
int compare2(const void* a, const void* b)
{
    return *(int*)b - *(int*)a;
}

int main() {
    int a[5] = { 1, 2, 5, 4, 3 };

    qsort(a, 5, sizeof(int), compare1);
    for (int i = 0;i < 5;i++) {
        cout << a[i] << " " ;
    }

    cout << endl;

    qsort(a, 5, sizeof(int), compare2);
    for (int i = 0;i < 5;i++) {
        cout << a[i] << " ";
    }
    return 0;
}
```

## 2.sort()：qsort()的改良版

（也更好用）

```c++
#include<algorithm>
//头文件
#include<utility>
#include<iostream>
using namespace std;

bool cmp(int a,int b)
{
    return a < b;
}
bool cmp(pair<int,int>a,pair<int,int>b)
{
    if(a.second != b.second)
        return a.second<b.second;
}

int main()
{
    int a[5] = { 1, 2, 5, 4, 3 };
    pair<int,int>a[] = {{1,2},{3,4}};
        //升序
    sort(a,a+2,cmp);
    for (int i = 0;i < 5;i++) {
        cout << a[i] << " ";
    }
    
    cout << endl;
    
        //降序
    sort(a,a+5,greater<int>());

        cout << a[i] << " ";
    }
    return 0;
}
```

## 4.去重unique()+erase()

1. 对象为vector等长度可变容器

```c++
#include <algorithm>

vector<int> v = {1, 2, 2, 3, 3, 4};

sort(v.begin(), v.end()); // 先排序
v.erase(unique(v.begin(), v.end()), v.end()); // 去重
```

1. 对象为静态数组

```c++
#include<algorithm>
//头文件

#include<iostream>
using namespace std;

int main()
{
    int a[] = { 1, 1, 2, 4, 5 };
    //1 2 4 5 *
    int* newlen = unique(a, a + 5);//这个函数的返回值是去重后最后一位有效数字的后一位的位置
    for (int* i = a;i < newlen;i++)//指针的++操作不是增加一个字节，而是增加sizeof(type)个字节
    {
        cout << *i << " ";
    }
    return 0;
}
```

## 6.二分查找lower_bound()/upper_bound()/binary_search()

```c++
#include<algorithm>
//头文件

#include<iostream>
using namespace std;

int main()
{
    int a[] = {0,1,2,2,4,5};
    int k = binary_search(a, a + 6, 0);//返回位置而非下标，找不到则返回0
    int pos1 = lower_bound(a, a + 6, 3) - a;//返回第一个大于等于目标值的地址
    int pos2 = upper_bound(a, a + 6, 3) - a;//返回第一个大于目标值的位置
    cout << k << endl;//1
    cout << pos1 << endl;
    cout << pos2 << endl;
    return 0;
}
```

## 8.数学函数

1. 绝对值
2. 对数函数
3. 二次方根
4. 向上取整
5. 向下取整
6. 四舍五入

```c++
#include <cmath>
//头文件

#include <iostream>
using namespace std;

int main() {
    cout << "abs(-10) = " << abs(-10) << endl;//绝对值
    cout << "pow(2, 3) = " << pow(2, 3) << endl;//2的3次方
    cout << "sqrt(16) = " << sqrt(16) << endl;//二次方根
    cout << "ceil(3.2) = " << ceil(3.2) << endl;//向上取整
    cout << "floor(3.8) = " << floor(3.8) << endl;//向下取整
    cout << "round(3.5) = " << round(3.5) << endl;//四舍五入
    cout << "log10(100) = " << log10(100) << endl;
    cout << "sin(π/2) = " << sin(3.14159 / 2) << endl;

    return 0;
}
```
