# 1.类与面向对象

什么是类？

```c++
#include <iostream>
#include <string>
using namespace std;

class Person {
private:
    string name;
    int age;

public:
    // 构造函数（用于初始化对象）
    Person(string Name, int Age) {
        name = Name;
        age = Age;
    }

    // 成员函数：打印信息
    void introduce() {
        cout << "Hello, my name is " << name << " and I am " << age << " years old." << endl;
    }

    // 更改年龄
    void setAge(int newAge) {
        if (newAge > 0) { // 确保年龄有效
            age = newAge;
        }
    }

    // 获取年龄
    int getAge() {
        return age;
    }
};

int main() {
    // 创建对象
    Person p1("Aleee", 20);
    Person p2("zy", 19);

    // 调用成员函数
    p1.introduce();
    p2.introduce();

    // 修改年龄
    person1.setAge(26);
    cout << "After one year, " << p1.getAge() << " years old." << endl;

    return 0;
}
```
