- 又名字典树，是哈希树的变种
- 用于统计，保存和排序大量的字符串（不仅限于）
- 基本性质：
	- 根节点：不保存数据（字符），可用于统计总数，包含 path 和 end
	- 路径：存放字符，为下一节点对应的字符串
	- 节点：存放 path 与 end ，若包含该字符则 path++，若到此字符字符串结束则 end++
	- 实现：使用类或 struct
	- 成员：`int end;` `Trie* path[];`
	- 初始化：`end = 0;` `memset(path ,0,sizeof(path));`

```c++
#include <iostream>
#include <unordered_map>
#include <string>
using namespace std;

struct TrieNode {
    int pass = 0;
    int end = 0;
    unordered_map<char, int> nexts;
};

// 全局数组模拟节点池
const int MAXN = 100000;
TrieNode trie[MAXN];
int nodeCnt = 1;  // 下标 0 是 root

void insert(const string& word) {
    int node = 0;
    trie[node].pass++;
    for (char c : word) {
        if (!trie[node].nexts.count(c)) {
            trie[node].nexts[c] = nodeCnt++;
        }
        node = trie[node].nexts[c];
        trie[node].pass++;
    }
    trie[node].end++;
}

void erase(const string& word) {
    if (!word.empty() && countWordsEqualTo(word) > 0) {
        int node = 0;
        trie[node].pass--;
        for (char c : word) {
            int next = trie[node].nexts[c];
            if (--trie[next].pass == 0) {
                trie[node].nexts.erase(c);
                return;
            }
            node = next;
        }
        trie[node].end--;
    }
}

int countWordsEqualTo(const string& word) {
    int node = 0;
    for (char c : word) {
        if (!trie[node].nexts.count(c)) return 0;
        node = trie[node].nexts[c];
    }
    return trie[node].end;
}

int countWordsStartingWith(const string& prefix) {
    int node = 0;
    for (char c : prefix) {
        if (!trie[node].nexts.count(c)) return 0;
        node = trie[node].nexts[c];
    }
    return trie[node].pass;
}

int main() {
    insert("apple");
    insert("apple");
    insert("app");
    cout << countWordsEqualTo("apple") << endl;        // 输出 2
    cout << countWordsStartingWith("app") << endl;     // 输出 3
    erase("apple");
    cout << countWordsEqualTo("apple") << endl;        // 输出 1
    cout << countWordsStartingWith("app") << endl;     // 输出 2
    erase("apple");
    cout << countWordsEqualTo("apple") << endl;        // 输出 0
    cout << countWordsStartingWith("app") << endl;     // 输出 1
    return 0;
}
```
